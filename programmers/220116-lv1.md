// js (하은)
function solution(absolutes, signs) {
    const answer = absolutes.reduce((acc, curr, idx, arr) => {
        let realNum = signs[idx] ? curr : -1 * curr;
        return acc + realNum
    }, 0)
    
    return answer;
}
